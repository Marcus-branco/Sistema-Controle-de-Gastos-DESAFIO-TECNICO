create table pessoa (id_pessoa int auto_increment,
nome varchar(150),
idade int(3),
primary key (id_pessoa)
);

create table despesa (id_despesa int auto_increment,
id_pessoa int,
descricao_despesa varchar(200),
valor_transacao float,
tipo_transacao varchar(100),
primary key (id_despesa)
);
